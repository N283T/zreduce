- [ ] **Step 3: Implement `generateMovers`**

The algorithm:
1. Iterate all atoms in the model
2. For atoms with `is_added` and `mover_hint != .none`, group by `(residue_idx, center_atom_name)`
3. For each group, look up the standard plan to find center and axis atom names
4. Resolve atom indices and call the appropriate constructor

Key implementation details:
- For `rotate` (single H): each H is its own mover. Center = plan.connected[0], axis = plan.connected[1].
- For `rotate_methyl` and `rotate_nh3`: 3 H atoms share the same center. Group by center atom name, then create one mover for the group.
- Skip `flip_amide` and `flip_his` (return without creating mover).

To find center/axis for a given H atom, look up its placement plan by matching comp_id + h_name against `standard.getPlans()`.

```zig
/// Find the placement plan for a given H atom by name and residue type.
fn findPlanForH(comp_id: []const u8, h_name: []const u8) ?*const standard.PlacementPlan {
    const plans = standard.getPlans(comp_id) orelse return null;
    for (plans) |*plan| {
        // Compare trimmed plan h_name with atom name
        var start: usize = 0;
        while (start < 4 and plan.h_name[start] == ' ') start += 1;
        var end: usize = 4;
        while (end > start and plan.h_name[end - 1] == ' ') end -= 1;
        if (std.mem.eql(u8, plan.h_name[start..end], h_name)) return plan;
    }
    return null;
}

/// Find a global atom index by name within a residue, searching both
/// original atoms and appended atoms (for placed H).
fn findAtomIdx(mdl: *const Model, residue_idx: u32, name: []const u8) ?u32 {
    for (mdl.atoms.items, 0..) |*a, i| {
        if (a.residue_idx == residue_idx and std.mem.eql(u8, a.nameSlice(), name)) {
            return @intCast(i);
        }
    }
    return null;
}

/// Trim a PDB 4-char name to its content (no leading/trailing spaces).
fn trimName(name: [4]u8) []const u8 {
    var start: usize = 0;
    while (start < 4 and name[start] == ' ') start += 1;
    var end: usize = 4;
    while (end > start and name[end - 1] == ' ') end -= 1;
    return name[start..end];
}

pub fn generateMovers(
    allocator: std.mem.Allocator,
    mdl: *const Model,
    no_flip: bool,
) ![]Mover {
    var movers = std.ArrayListUnmanaged(Mover){};
    errdefer {
        for (movers.items) |*m| m.deinit();
        movers.deinit(allocator);
    }

    // Track which groups have been processed to avoid duplicates.
    // Key: (residue_idx, center_atom_name_hash) → already processed
    var processed = std.AutoHashMapUnmanaged(u64, void){};
    defer processed.deinit(allocator);

    for (mdl.atoms.items, 0..) |atom, atom_i| {
        if (!atom.is_added or !atom.is_hydrogen) continue;
        if (atom.mover_hint == .none) continue;

        // Skip flippers for now
        if (atom.mover_hint == .flip_amide or atom.mover_hint == .flip_his) continue;
        _ = no_flip; // will be used for flip suppression in Issue #17

        const res = mdl.residues.items[atom.residue_idx];
        const comp_id = res.compIdSlice();
        const h_name = atom.nameSlice();

        // Find the placement plan for this H
        const plan = findPlanForH(comp_id, h_name) orelse continue;
        const center_name = trimName(plan.connected[0]);
        const axis_name = trimName(plan.connected[1]);

        // For group movers (methyl, NH3), use dedup key
        if (atom.mover_hint == .rotate_methyl or atom.mover_hint == .rotate_nh3) {
            const center_idx = findAtomIdx(mdl, atom.residue_idx, center_name) orelse continue;
            const key: u64 = @as(u64, atom.residue_idx) << 32 | @as(u64, center_idx);
            if (processed.contains(key)) continue;
            try processed.put(allocator, key, {});

            const axis_idx = findAtomIdx(mdl, atom.residue_idx, axis_name) orelse continue;

            // Find all 3 H atoms for this group
            var h_indices: [3]u32 = undefined;
            var h_count: usize = 0;
            const plans = standard.getPlans(comp_id) orelse continue;
            for (plans) |*p| {
                if (p.mover_hint != atom.mover_hint) continue;
                if (!std.mem.eql(u8, trimName(p.connected[0]), center_name)) continue;
                const p_h_name = trimName(p.h_name);
                if (findAtomIdx(mdl, atom.residue_idx, p_h_name)) |idx| {
                    if (h_count < 3) {
                        h_indices[h_count] = idx;
                        h_count += 1;
                    }
                }
            }
            if (h_count != 3) continue; // incomplete group, skip

            const m = if (atom.mover_hint == .rotate_methyl)
                try rotator.createMethylRotator(allocator, mdl.atoms.items, h_indices, center_idx, axis_idx, atom.residue_idx)
            else
                try rotator.createNH3Rotator(allocator, mdl.atoms.items, h_indices, center_idx, axis_idx, atom.residue_idx);
            try movers.append(allocator, m);
        } else if (atom.mover_hint == .rotate) {
            // Single H rotator
            const h_idx: u32 = @intCast(atom_i);
            const center_idx = findAtomIdx(mdl, atom.residue_idx, center_name) orelse continue;
            const axis_idx = findAtomIdx(mdl, atom.residue_idx, axis_name) orelse continue;

            const m = try rotator.createSingleHRotator(allocator, mdl.atoms.items, h_idx, center_idx, axis_idx, atom.residue_idx);
            try movers.append(allocator, m);
        }
    }

    const result = try allocator.dupe(Mover, movers.items);
    movers.deinit(allocator);
    return result;
}
```
