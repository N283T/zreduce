**Goal:** Make hydrogen placement conformer-aware so multi-conformer residues use correct per-conformer geometry, not mixed-altloc positions.

**Architecture:** Add `target_altloc: u8` parameter to all atom lookup functions (`findAtomPos`, `findAtom`, and all functions that call them). Implement a two-pass lookup: prefer exact altloc match, fall back to blank altloc (shared atoms). Add conformer iteration in `addHydrogens` so each conformer gets its own placement pass. This is a mechanical signature change propagated through ~10 functions.

**Tech Stack:** Zig, zig test

/// Find an atom by 4-char PDB name within a residue, preferring the given altloc.
/// Fallback: if no exact altloc match, returns atom with blank altloc (' ').
fn findAtomPos(mdl: *const Model, res: Residue, name: [4]u8, target_altloc: u8) ?Vec3f32 {
    if (isBlank(name)) return null;
    const atoms = mdl.atoms.items[res.atom_start..res.atom_end];
    var blank_match: ?Vec3f32 = null;
    for (atoms) |a| {
        if (nameMatch(name, a.nameSlice())) {
            if (a.altloc == target_altloc) return a.pos;
            if (a.altloc == ' ' and blank_match == null) blank_match = a.pos;
        }
    }
    return blank_match;
}
```

- [ ] **Step 3: Add `collectAltlocs` helper**

/// Collect distinct non-blank altloc values from a residue's atoms.
/// Returns a fixed-size array with up to 10 distinct altlocs and the count.
fn collectAltlocs(mdl: *const Model, res: Residue) struct { locs: [10]u8, count: usize } {
    var result = .{ .locs = [_]u8{0} ** 10, .count = @as(usize, 0) };
    const atoms = mdl.atoms.items[res.atom_start..res.atom_end];
    for (atoms) |a| {
        if (a.altloc == ' ') continue;
        // Check if already collected
        var found = false;
        for (result.locs[0..result.count]) |existing| {
            if (existing == a.altloc) { found = true; break; }
        }
        if (!found and result.count < 10) {
            result.locs[result.count] = a.altloc;
            result.count += 1;
        }
    }
    return result;
}
```

- [ ] **Step 4: Update `addHydrogens` with conformer loop**

In the standard-residue branch, replace the direct plan loop with conformer-aware iteration:

```zig
if (standard.getPlans(comp_id)) |plans| {
    const bonds = topology.getBonds(comp_id);
    const altlocs = collectAltlocs(mdl, res);

    if (altlocs.count == 0) {
        // Single conformer (all blank altloc) — process once
        for (plans) |plan| {
            if (is_nterm and isBackboneH(&plan)) continue;

            if (try executePlan(mdl, res, @intCast(res_idx), &plan, bonds, ' ')) {
                result.n_placed += 1;
            } else {
                result.n_skipped += 1;
            }
        }

        if (is_nterm) {
            const nterm = if (std.mem.eql(u8, comp_id, "PRO"))
                try placeNtermNH2Pro(mdl, res, @intCast(res_idx), ' ')
            else
                try placeNtermNH3(mdl, res, @intCast(res_idx), ' ');
            result.n_placed += nterm.placed;
            result.n_skipped += nterm.skipped;
        }
    } else {
        // Multi-conformer — process once per conformer
        for (altlocs.locs[0..altlocs.count]) |alt| {
            for (plans) |plan| {
                if (is_nterm and isBackboneH(&plan)) continue;

                if (try executePlan(mdl, res, @intCast(res_idx), &plan, bonds, alt)) {
                    result.n_placed += 1;
                } else {
                    result.n_skipped += 1;
                }
            }

            if (is_nterm) {
                const nterm = if (std.mem.eql(u8, comp_id, "PRO"))
                    try placeNtermNH2Pro(mdl, res, @intCast(res_idx), alt)
                else
                    try placeNtermNH3(mdl, res, @intCast(res_idx), alt);
                result.n_placed += nterm.placed;
                result.n_skipped += nterm.skipped;
            }
        }
    }

    result.n_residues += 1;
}
```
