# Issue #15: Residue-Specific Chemical Typing — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Assign residue/atom-specific AtomType and AtomFlags to standard-residue heavy atoms so scoring can correctly identify H-bond donors, acceptors, aromatic atoms, and charged groups.

**Architecture:** New file `src/place/chemistry.zig` provides a comptime annotation table keyed by `(comp_id, atom_name)`. A new `applyChemistry` function iterates all standard-residue atoms and applies annotations. `appendHydrogen` is fixed to propagate flags from the element table. The function is called from `main.zig` before hydrogen placement.

**Tech Stack:** Zig, zig test

```zig
const A = element.AtomFlags{ .acceptor = true };
const D = element.AtomFlags{ .donor = true };
const AR = element.AtomFlags{ .aromatic = true };
const ARA = element.AtomFlags{ .aromatic = true, .acceptor = true };
const NEG_A = element.AtomFlags{ .negative = true, .acceptor = true };
const POS_D = element.AtomFlags{ .positive = true, .donor = true };

const Entry = struct { name: [4]u8, ann: ChemAnnotation };

fn e(comptime name: []const u8, atom_type: element.AtomType, flags: element.AtomFlags) Entry {
    return .{ .name = n(name), .ann = .{ .atom_type = atom_type, .flags = flags } };
}

// Backbone annotations (apply to all standard residues)
const backbone_annotations = [_]Entry{
    e(" C  ", .C_eq_O, .{}),    // carbonyl carbon, smaller radius
    e(" O  ", .O, A),           // carbonyl oxygen, acceptor
    e(" N  ", .N, D),           // amide nitrogen, donor
};

// --- Side-chain annotations per residue ---

const asp_annotations = [_]Entry{
    e("OD1 ", .O, NEG_A),
    e("OD2 ", .O, NEG_A),
    e(" CG ", .C_eq_O, .{}),
};

const glu_annotations = [_]Entry{
    e("OE1 ", .O, NEG_A),
    e("OE2 ", .O, NEG_A),
    e(" CD ", .C_eq_O, .{}),
};

const asn_annotations = [_]Entry{
    e("OD1 ", .O, A),
    e(" CG ", .C_eq_O, .{}),
    e("ND2 ", .N, D),
};

const gln_annotations = [_]Entry{

};

const lys_annotations = [_]Entry{

const arg_annotations = [_]Entry{

    e("NH1 ", .N, POS_D),
    e("NH2 ", .N, POS_D),
    e(" CZ ", .C_eq_O, .{}),  // sp2 carbon in guanidinium
};

const ser_annotations = [_]Entry{
    e(" OG ", .O, A),
};

const thr_annotations = [_]Entry{
    e("OG1 ", .O, A),
};

const tyr_annotations = [_]Entry{
    e(" OH ", .O, A),
    // Ring carbons
    e(" CG ", .Car, ARA),
    e("CD1 ", .Car, ARA),
    e("CD2 ", .Car, ARA),
    e("CE1 ", .Car, ARA),
    e("CE2 ", .Car, ARA),
    e(" CZ ", .Car, ARA),
};

const phe_annotations = [_]Entry{
    e(" CG ", .Car, ARA),
    e("CD1 ", .Car, ARA),
    e("CD2 ", .Car, ARA),
    e("CE1 ", .Car, ARA),
    e("CE2 ", .Car, ARA),
    e(" CZ ", .Car, ARA),
};

    e(" CG ", .Car, ARA),
    e("CD1 ", .Car, ARA),
    e("CD2 ", .Car, ARA),
    e("NE1 ", .N, .{ .aromatic = true, .donor = true }),
    e("CE2 ", .Car, ARA),
    e("CE3 ", .Car, ARA),
    e("CZ2 ", .Car, ARA),
    e("CZ3 ", .Car, ARA),
    e("CH2 ", .Car, ARA),
};

const his_annotations = [_]Entry{
    e(" CG ", .Car, ARA),
    e("CD2 ", .Car, ARA),
    e("CE1 ", .Car, ARA),
    e("ND1 ", .Nacc, ARA),    // acceptor in default tautomer
    e("NE2 ", .Nacc, ARA),    // acceptor in default tautomer
};

const cys_annotations = [_]Entry{
    e(" SG ", .S, A),
};

const met_annotations = [_]Entry{
    e(" SD ", .S, A),
};

const empty_annotations = [_]Entry{};

fn getSideChainAnnotations(comp_id: []const u8) ?[]const Entry {
    const map = std.StaticStringMap([]const Entry).initComptime(.{
        .{ "ASP", &asp_annotations },
        .{ "GLU", &glu_annotations },
        .{ "ASN", &asn_annotations },
        .{ "GLN", &gln_annotations },
        .{ "LYS", &lys_annotations },
        .{ "ARG", &arg_annotations },
        .{ "SER", &ser_annotations },
        .{ "THR", &thr_annotations },
        .{ "TYR", &tyr_annotations },
        .{ "PHE", &phe_annotations },
        .{ "TRP", &trp_annotations },
        .{ "HIS", &his_annotations },
        .{ "CYS", &cys_annotations },
        .{ "MET", &met_annotations },
        // Residues with no side-chain annotations:
        .{ "ALA", &empty_annotations },
        .{ "GLY", &empty_annotations },
        .{ "VAL", &empty_annotations },
        .{ "LEU", &empty_annotations },
        .{ "ILE", &empty_annotations },
        .{ "PRO", &empty_annotations },
    });
    return map.get(comp_id);
}

pub fn getAnnotation(comp_id: []const u8, atom_name: [4]u8) ?ChemAnnotation {
    // Must be a known standard residue
    const side_chain = getSideChainAnnotations(comp_id) orelse return null;

    // Check side-chain annotations first (residue-specific)
    for (side_chain) |entry| {
        if (std.mem.eql(u8, &entry.name, &atom_name)) return entry.ann;
    }

    // Check backbone annotations (shared)
    for (backbone_annotations) |entry| {
        if (std.mem.eql(u8, &entry.name, &atom_name)) return entry.ann;
    }

    return null;
}
```

```zig
const chemistry = @import("chemistry.zig");

/// Apply residue/atom-specific chemical annotations to heavy atoms.
/// Updates element_type, flags, and vdw_radius on standard-residue heavy atoms.
/// Should be called once after parsing, before hydrogen placement.
pub fn applyChemistry(mdl: *Model) void {
    for (mdl.residues.items) |res| {
        const comp_id = res.compIdSlice();
        const atoms = mdl.atoms.items[res.atom_start..res.atom_end];
        for (atoms) |*atom| {
            if (atom.is_hydrogen) continue;
            if (chemistry.getAnnotation(comp_id, atom.name)) |ann| {
                atom.element_type = ann.atom_type;
                atom.flags = ann.flags;
                atom.vdw_radius = ann.atom_type.info().explicit_radius;
            }
        }
    }
}
```
