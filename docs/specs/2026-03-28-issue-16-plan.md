/// Generate fine orientations around a coarse best orientation.
/// For single_h_rotator: ±15° at 5° steps (6 positions).
/// For methyl/nh3: ±30° at 10° steps (6 positions).
/// Returns temporary orientations that caller must free.
pub fn generateFineOrientations(
    allocator: std.mem.Allocator,
    atoms: []const Atom,
    m: *const Mover,
    coarse_best_idx: u16,
) ![]Orientation {
    const center_idx = m.center_idx orelse return &.{};
    const axis_idx = m.axis_idx orelse return &.{};

    const center = atoms[center_idx].pos;
    const axis_atom = atoms[axis_idx].pos;
    const axis_dir = center.sub(axis_atom);

    // Determine base angle from coarse best index
    const base_angle: f32 = switch (m.kind) {
        .single_h_rotator => @as(f32, @floatFromInt(coarse_best_idx)) * 30.0,
        .methyl_rotator, .nh3_rotator, .aromatic_methyl => blk: {
            const offsets = [3]f32{ 0.0, 60.0, -60.0 };
            break :blk if (coarse_best_idx < 3) offsets[coarse_best_idx] else 0.0;
        },
        else => return &.{},
    };

    // Fine search parameters
    const half_range: f32 = switch (m.kind) {
        .single_h_rotator => 15.0,
        .methyl_rotator, .nh3_rotator, .aromatic_methyl => 30.0,
        else => return &.{},
    };
    const step: f32 = switch (m.kind) {
        .single_h_rotator => 5.0,
        .methyl_rotator, .nh3_rotator, .aromatic_methyl => 10.0,
        else => return &.{},
    };

    // Count fine steps (exclude 0 offset — that's the coarse best)
    var n_fine: usize = 0;
    var angle = -half_range;
    while (angle <= half_range) : (angle += step) {
        if (@abs(angle) < 0.1) continue; // skip center (coarse best)
        n_fine += 1;
    }

    const orientations = try allocator.alloc(Orientation, n_fine);
    var oi: usize = 0;
    angle = -half_range;
    while (angle <= half_range) : (angle += step) {
        if (@abs(angle) < 0.1) continue;
        const fine_angle = base_angle + angle;
        const n_atoms = m.atom_indices.len;
        const positions = try allocator.alloc(Vec3(f32), n_atoms);
        for (0..n_atoms) |j| {
            // Use the ORIGINAL atom position (orientation 0) as reference
            const orig_pos = m.orientations[0].positions[j];
            positions[j] = math_mod.rotateAroundAxis(f32, orig_pos, center, axis_dir, fine_angle);
        }
        orientations[oi] = .{ .positions = positions, .penalty = 0.0 };
        oi += 1;
    }

    return orientations;
}
```

/// Fine-search a single mover around its current best coarse orientation.
/// Generates fine orientations, scores them, and updates best_orientation
/// if a fine position improves the score. The mover's orientations array
/// is extended with the best fine orientation if one is found.
fn fineSearchMover(
    allocator: Allocator,
    movers: []Mover,
    mover_idx: u32,
    model: *Model,
    config: OptConfig,
) void {
    const m = &movers[mover_idx];
    const atoms = model.atoms.items;

    // Generate fine orientations around coarse best
    const fine = rotator_mod.generateFineOrientations(allocator, atoms, m, m.best_orientation) catch return;
    defer {
        for (fine) |o| allocator.free(o.positions);
        allocator.free(fine);
    }

    if (fine.len == 0) return;

    // Score current coarse best
    m.applyOrientation(atoms, m.best_orientation);
    var best_score = scoreMover(m, mover_idx, movers, model, config) - m.orientationPenalty(m.best_orientation);

    // Score fine orientations
    var best_fine: ?usize = null;
    for (fine, 0..) |orient, fi| {
        // Apply fine orientation manually
        for (m.atom_indices, 0..) |ai, j| {
            atoms[ai].pos = orient.positions[j];
        }

        const score = scoreMover(m, mover_idx, movers, model, config) - orient.penalty;
        if (score > best_score) {
            best_score = score;
            best_fine = fi;
        }
    }

    if (best_fine) |fi| {
        // Apply the best fine orientation permanently
        for (m.atom_indices, 0..) |ai, j| {
            atoms[ai].pos = fine[fi].positions[j];
        }
    } else {
        // Restore coarse best
        m.applyOrientation(atoms, m.best_orientation);
    }
}
```

Note: This modifies atom positions directly rather than extending the orientations array. The fine search is a refinement step — the result is baked into atom positions.

test "iterative greedy improves over independent greedy for coupled movers" {
    // Two movers that interact: best solution requires considering both.
    // Mover 0: atom 1, orientations at (2,0,0) and (0,2,0)
    // Mover 1: atom 2, orientations at (2.5,0,0) and (0,2.5,0)
    // Fixed obstacle at (0,0,0) with radius 1.7
    //
    // If both pick orient 0: atom1 at (2,0,0) and atom2 at (2.5,0,0) — they're close (0.5 apart)
    //   but atom1 is far from obstacle (dist 2.0 > sum_r 3.4? no, 2.0 < 3.4 → slight bump)
    // If mover0=orient1, mover1=orient1: atoms far apart and from obstacle → best
    //
    // Independent greedy might pick different solutions depending on order.
    // Iterative should converge to (1,1) which avoids all interactions.
    const allocator = testing.allocator;

    var model = Model.init(allocator);
    defer model.deinit();

    try model.atoms.append(allocator, .{
        .pos = .{ .x = 0, .y = 0, .z = 0 },
        .vdw_radius = 1.7,
    });
    try model.atoms.append(allocator, .{
        .pos = .{ .x = 5, .y = 0, .z = 0 },
        .vdw_radius = 1.0,
    });
    try model.atoms.append(allocator, .{
        .pos = .{ .x = 0, .y = 5, .z = 0 },
        .vdw_radius = 1.0,
    });

    var m0 = try makeTestMover(allocator, 1, &.{
        .{ .x = 2.0, .y = 0, .z = 0 },
        .{ .x = 0, .y = 10, .z = 0 },
    }, &.{ 0, 0 });
    defer m0.deinit();

    var m1 = try makeTestMover(allocator, 2, &.{
        .{ .x = 2.5, .y = 0, .z = 0 },
        .{ .x = 0, .y = 10.5, .z = 0 },
    }, &.{ 0, 0 });
    defer m1.deinit();

    var movers = [_]Mover{ m0, m1 };
    const clq = [_]u32{ 0, 1 };

    optimizeIterativeGreedy(&movers, &clq, &model, .{});

    // Both should pick orientation 1 (far from obstacle)
    try testing.expectEqual(@as(u16, 1), movers[0].best_orientation);
    try testing.expectEqual(@as(u16, 1), movers[1].best_orientation);
}

fn optimizeIterativeGreedy(movers: []Mover, clq: []const u32, model: *Model, config: OptConfig) void {
    const max_iterations: u32 = 3;
    var iteration: u32 = 0;

    while (iteration < max_iterations) : (iteration += 1) {
        var changed = false;
        for (clq) |mi| {
            const old_best = movers[mi].best_orientation;
            optimizeSingleton(movers, mi, model, config);

            // Apply the new best so subsequent movers see updated positions
            movers[mi].applyOrientation(model.atoms.items, movers[mi].best_orientation);

            if (movers[mi].best_orientation != old_best) changed = true;
        }
        if (!changed) break;
    }
}
